// 전역변수 Logger_v 직접 생성 및 설정해주면 된다. */
Logger_v = dbg_new ConsoleLogger();

Logger_v->SetAutoFlush(true);

Logger_v->ShowLevel(true);

// 종료전 삭제
delete Logger_v;
